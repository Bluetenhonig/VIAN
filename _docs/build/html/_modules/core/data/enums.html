def get_type_as_string(type):
    if type == PROJECT:
        return "PROJECT"
    elif type == SEGMENTATION:
        return "SEGMENTATION"
    elif type == SEGMENT:
        return "SEGMENT"
    elif type == ANNOTATION:
        return "ANNOTATION"

def get_enum_value(enum, name):
    for s in enum:
        if name == s.name:
            return s.value


def get_enum(enum, name):
    for s in enum:
        if name == s.name:
            return s
